    95→  const [, startTransitionLogin] = useTransition();

    97→    fetch('/api/users').then(r => r.json()).then(data => {

    99→    });

   114→  };

   343→function Dashboard() {

   348→  const [notifs, setNotifs] = useState<Notification[]>([]);
   349→
   350→  useEffect(() => {
   351→    Promise.all([
   352→      fetch('/api/stats').then(r => r.json()),

   354→      fetch('/api/trucks').then(r => r.json()),
   355→      fetch('/api/notifications').then(r => r.json()),
   356→    ]).then(([s, c, t, n]) => { setStats(s); setRecentCargo(c); setTrucks(t); setNotifs(n); });

   358→

   479→function ShiftManagement() {
   480→  const { currentUser } = useAppStore();
   481→  const [shifts, setShifts] = useState<Shift[]>([]);
   482→  const [activeShift, setActiveShift] = useState<Shift | null>(null);
   483→  const [checkInDialog, setCheckInDialog] = useState(false);
   484→  const [newShift, setNewShift] = useState({ type: 'day', location: '' });
   485→
   486→  const shiftsLoadedRef = useRef(false);
   487→  const [, startTransition] = useTransition();
   488→  const loadShifts = useCallback(async () => {
   489→    if (!currentUser) return;
   490→    const [allShifts, myShifts] = await Promise.all([
   491→      fetch('/api/shifts').then(r => r.json()),
   492→      fetch(`/api/shifts?userId=${currentUser.id}&status=active`).then(r => r.json()),
   493→    ]);
   494→    startTransition(() => {
   495→      setShifts(allShifts);
   496→      setActiveShift(myShifts.length > 0 ? myShifts[0] : null);
   497→    });
   498→  }, [currentUser, startTransition]);
   499→
   500→  useEffect(() => {
   501→    if (currentUser && !shiftsLoadedRef.current) {
   502→      shiftsLoadedRef.current = true;
   503→      loadShifts();
   504→    }
   505→  }, [currentUser, loadShifts]);
   506→
   507→  const checkIn = async () => {
   508→    if (!currentUser) return;
   509→    const res = await fetch('/api/shifts', {
   510→      method: 'POST',
   511→      headers: { 'Content-Type': 'application/json' },
   512→      body: JSON.stringify({ userId: currentUser.id, type: newShift.type, location: newShift.location, status: 'active' }),
   513→    });
   514→    if (res.ok) { toast({ title: 'Checked In', description: `Shift started at ${newShift.location}` }); setCheckInDialog(false); loadShifts(); }
   515→  };
   516→
   517→  const checkOut = async () => {
   518→    if (!activeShift) return;
   519→    const res = await fetch('/api/shifts', {
   520→      method: 'PATCH',
   521→      headers: { 'Content-Type': 'application/json' },
   522→      body: JSON.stringify({ id: activeShift.id, status: 'ended', checkOut: new Date().toISOString() }),
   523→    });
   524→    if (res.ok) { toast({ title: 'Checked Out', description: 'Shift ended' }); loadShifts(); }
   525→  };
   526→
   527→  const startBreak = async () => {
   528→    if (!activeShift) return;
   529→    const res = await fetch('/api/shifts', { method: 'PATCH', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ id: activeShift.id, status: 'break' }) });
   530→    if (res.ok) { toast({ title: 'Break Started' }); loadShifts(); }
   531→  };
   532→
   533→  const endBreak = async () => {
   534→    if (!activeShift) return;
   535→    const res = await fetch('/api/shifts', { method: 'PATCH', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ id: activeShift.id, status: 'active' }) });
   536→    if (res.ok) { toast({ title: 'Break Ended' }); loadShifts(); }
   537→  };
   538→
   539→  return (
   540→    <div className="p-4 lg:p-6 space-y-6 max-w-4xl mx-auto">
   541→      {/* Active Shift Card */}
   542→      <Card className="overflow-hidden border-0 shadow-lg">
   543→        {activeShift ? (
   544→          <div className="bg-gradient-to-r from-green-500 to-emerald-600 p-6 text-white">
   545→            <div className="flex items-center justify-between">
   546→              <div>
   547→                <p className="text-green-100 text-sm font-medium">Active Shift</p>
   548→                <p className="text-2xl font-bold mt-1">{activeShift.type === 'day' ? 'Day' : 'Night'} Shift</p>
   549→                <div className="flex items-center gap-4 mt-2 text-green-100 text-sm">
   550→                  <span className="flex items-center gap-1"><Clock className="h-4 w-4" /> {formatTime(activeShift.checkIn)}</span>
   551→                  <span className="flex items-center gap-1"><MapPin className="h-4 w-4" /> {activeShift.location || 'Not assigned'}</span>
   552→                </div>
   553→              </div>
   554→              <div className="text-right">
   555→                <Badge className="bg-white/20 text-white border-white/30 text-sm">{activeShift.status}</Badge>
   556→                <p className="text-green-100 text-sm mt-2">Since {timeSince(activeShift.checkIn)}</p>
   557→              </div>
   558→            </div>
   559→            <div className="flex gap-3 mt-6">
   560→              {activeShift.status === 'active' ? (
   561→                <Button onClick={startBreak} variant="secondary" className="bg-white/20 hover:bg-white/30 text-white border-0">
   562→                  <Pause className="h-4 w-4 mr-2" /> Break
   563→                </Button>
   564→              ) : (
   565→                <Button onClick={endBreak} variant="secondary" className="bg-white/20 hover:bg-white/30 text-white border-0">
   566→                  <Play className="h-4 w-4 mr-2" /> Resume
   567→                </Button>
   568→              )}
   569→              <Button onClick={checkOut} variant="secondary" className="bg-red-500/80 hover:bg-red-500 text-white border-0">
   570→                <Square className="h-4 w-4 mr-2" /> Check Out
   571→              </Button>
   572→            </div>
   573→          </div>
   574→        ) : (
   575→          <div className="bg-gradient-to-r from-slate-600 to-slate-700 p-6 text-white text-center">
   576→            <Timer className="h-12 w-12 mx-auto mb-3 opacity-50" />
   577→            <p className="text-lg font-semibold">No Active Shift</p>
   578→            <p className="text-slate-300 text-sm mt-1">Check in to start your shift</p>
   579→            <Dialog open={checkInDialog} onOpenChange={setCheckInDialog}>
   580→              <DialogTrigger asChild>
   581→                <Button className="mt-4 bg-gradient-to-r from-amber-500 to-orange-600 hover:from-amber-600 hover:to-orange-700 text-white">
   582→                  <Clock className="h-4 w-4 mr-2" /> Check In
   583→                </Button>
   584→              </DialogTrigger>
   585→              <DialogContent>
   586→                <DialogHeader><DialogTitle>Check In for Shift</DialogTitle></DialogHeader>
   587→                <div className="space-y-4">
   588→                  <div><Label>Shift Type</Label>
   589→                    <Select value={newShift.type} onValueChange={(v) => setNewShift(p => ({ ...p, type: v }))}>
   590→                      <SelectTrigger><SelectValue /></SelectTrigger>
   591→                      <SelectContent><SelectItem value="day">Day Shift (06:00-14:00)</SelectItem><SelectItem value="night">Night Shift (14:00-22:00)</SelectItem></SelectContent>
   592→                    </Select>
   593→                  </div>
   594→                  <div><Label>Location / Dock</Label>
   595→                    <Input placeholder="e.g. Quai 125, Berth 127" value={newShift.location} onChange={(e) => setNewShift(p => ({ ...p, location: e.target.value }))} />
   596→                  </div>
   597→                </div>
   598→                <DialogFooter><Button onClick={checkIn} className="bg-gradient-to-r from-green-500 to-emerald-600 text-white">Start Shift</Button></DialogFooter>
   599→              </DialogContent>
   600→            </Dialog>
   601→          </div>
   602→        )}
   603→      </Card>
   604→
   605→      {/* Shift History */}
   606→      <Card>
   607→        <CardHeader><CardTitle className="text-lg">Recent Shifts</CardTitle></CardHeader>
   608→        <CardContent>
   609→          <div className="space-y-2">
   610→            {shifts.map(s => (
   611→              <div key={s.id} className="flex items-center gap-3 p-3 rounded-lg bg-slate-50 dark:bg-slate-800/50">
   612→                <div className={`w-2 h-2 rounded-full ${s.status === 'active' ? 'bg-green-500' : s.status === 'break' ? 'bg-purple-500' : 'bg-gray-400'}`} />

   615→                  <p className="text-xs text-muted-foreground">{s.type} shift — {s.location || 'No location'}</p>
   616→                </div>
   617→                <div className="text-right">
   618→                  <Badge className={statusColors[s.status] || ''} variant="outline">{s.status}</Badge>
   619→                  <p className="text-xs text-muted-foreground mt-1">{formatTime(s.checkIn)} — {s.checkOut ? formatTime(s.checkOut) : '...'}</p>

   621→              </div>
   622→            ))}
   623→          </div>
